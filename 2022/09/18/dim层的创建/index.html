1. 商品维度表（全量）

建表语句

DROP TABLE IF EXISTS dim_sku_info;
CREATE EXTERNAL TABLE dim_sku_info (
    `id` STRING COMMENT '商品id',
    `price` DECIMAL(16,2) COMMENT '商品价格',
    `sku_name` STRING COMMENT '商品名称',
    `sku_desc` STRING COMMENT '商品描述',
    `weight` DECIMAL(16,2) COMMENT '重量',
    `is_sale` BOOLEAN COMMENT '是否在售',
    `spu_id` STRING COMMENT 'spu编号',
    `spu_name` STRING COMMENT 'spu名称',
    `category3_id` STRING COMMENT '三级分类id',
    `category3_name` STRING COMMENT '三级分类名称',
    `category2_id` STRING COMMENT '二级分类id',
    `category2_name` STRING COMMENT '二级分类名称',
    `category1_id` STRING COMMENT '一级分类id',
    `category1_name` STRING COMMENT '一级分类名称',
    `tm_id` STRING COMMENT '品牌id',
    `tm_name` STRING COMMENT '品牌名称',
    `sku_attr_values` ARRAY<STRUCT<attr_id:STRING,value_id:STRING,attr_name:STRING,value_name:STRING>> COMMENT '平台属性',

    `create_time` STRING COMMENT '创建时间'
) COMMENT '商品维度表'
    PARTITIONED BY (`dt` STRING)
    STORED AS ORC
    LOCATION '/warehouse/gmall/dim/dim_sku_info/'
    TBLPROPERTIES ("orc.compress"="snappy");

查询插入语句

这里使用了一个with … as用法，写起来比较容易理解

named_struct函数，可以把一行的多个属性值合并成为一个结构体

collect_list函数可以把多行转换成一个array数据

with sku_info as (
    select
        id,
        spu_id,
        price,
        sku_name,
        sku_desc,
        weight,
        tm_id,
        category3_id,
        is_sale,
        create_time
    from ods_sku_info
    where dt = '2022-09-01'
    ),
    spu_info as (
    select
        id,
        spu_name
    from ods_spu_info
    where dt = '2022-09-01'
    ),
    tm as (
        select
            id,
            tm_name
        from ods_base_trademark
        where dt = '2022-09-01'
    ),
    cate_3 as (
        select
            id,
            name,
            category2_id
        from ods_base_category3
        where dt = '2022-09-01'
    ),
    cate_2 as (
        select
            id,
            name,
            category1_id
        from ods_base_category2
        where dt = '2022-09-01'
    ),
    cate_1 as (
        select
            id,
            name
        from ods_base_category1
        where dt = '2022-09-01'
    ),
    sku_attr_value as (
        select
            sku_id,
            collect_list(
                    named_struct(
                            'attr_id',attr_id,
                            'value_id',value_id,
                            'attr_name',attr_name,
                            'value_name',value_name
                        )
                ) as value1
        from ods_sku_attr_value
        where dt = '2022-09-01'
        group by sku_id
    ),
    sku_sale_value as (
        select
            sku_id,
            collect_list(
                named_struct(
                        'sale_attr_id',sale_attr_id,
                        'sale_attr_value_id',sale_attr_value_id,
                        'sale_attr_name',sale_attr_name,
                        'sale_attr_value_name',sale_attr_value_name
                    )
                ) as value1
        from ods_sku_sale_attr_value
        where dt = '2022-09-01'
        group by sku_id
    )
insert overwrite table dim_sku_info partition (dt='2022-09-01')
select
    sku_info.id,
    price,
    sku_name,
    sku_desc,
    weight,
    is_sale,
    spu_id,
    spu_name,
    category3_id,
    cate_3.name,
    category2_id,
    cate_2.name,
    category1_id,
    cate_1.name,
    tm_id,
    tm_name,
    sku_attr_value.value1,
    sku_sale_value.value1,
    create_time
from sku_info
left join spu_info on sku_info.spu_id = spu_info.id
left join tm on tm_id = tm.id
left join cate_3 on category3_id = cate_3.id
left join cate_2 on category2_id = cate_2.id
left join cate_1 on category1_id = cate_1.id
left join sku_attr_value on sku_info.id = sku_attr_value.sku_id
left join sku_sale_value on sku_info.id = sku_sale_value.sku_id

insert overwrite table dim_activity_rule_info partition(dt='2022-09-01')
select
    ar.id,
    ar.activity_id,
    ai.activity_name,
    ar.activity_type,
    ai.start_time,
    ai.end_time,
    ai.create_time,
    ar.condition_amount,
    ar.condition_num,
    ar.benefit_amount,
    ar.benefit_discount,
    ar.benefit_level
from
    (
        select
            id,
            activity_id,
            activity_type,
            condition_amount,
            condition_num,
            benefit_amount,
            benefit_discount,
            benefit_level
        from ods_activity_rule
        where dt='2022-09-01'
    )ar
        left join
    (
        select
            id,
            activity_name,
            start_time,
            end_time,
            create_time
        from ods_activity_info
        where dt='2022-09-01'
    )ai
    on ar.activity_id=ai.id;

每日装载

本部分有点麻烦，首先我们需要把当天的数据和9999的join起来，如果数据被修改了就放入昨天的分区中，如果新增以及没有被修改的就放入9999分区中。

可以有两种思路来实现，一种是写两个sql，一种是使用动态分区来插入。

#!/bin/bash

APP=gmall

# 如果是输入的日期按照取输入日期；如果没输入日期取当前时间的前一天
if [ -n "$2" ] ;then
do_date=$2
seconds=`date -d $2 +%s`
seconds_yesterday=$((seconds - 86400))
day_before=`date -d @$seconds_yesterday +%F`
else
    exit 
    do_date=`date -d "-1 day" +%F`
fi
dim_user_info="
with
    tmp as
        (
            select

dim_sku_info="
with
sku as
(
    select
        id,
        price,
        sku_name,
        sku_desc,
        weight,
        is_sale,
        spu_id,

    where dt='$do_date'
),
spu as

        spu_name
    from ${APP}.ods_spu_info
    where dt='$do_date'
),
c3 as
(

        id,
        name,
        category2_id
    from ${APP}.ods_base_category3
    where dt='$do_date'
),
c2 as
(

    from ${APP}.ods_base_category2
    where dt='$do_date'
),
c1 as
(
    select
        id,
        name
    from ${APP}.ods_base_category1
    where dt='$do_date'
),
tm as
(
    select
        id,
        tm_name
    from ${APP}.ods_base_trademark
    where dt='$do_date'
),
attr as
(
    select
        sku_id,
        collect_set(named_struct('attr_id',attr_id,'value_id',value_id,'attr_name',attr_name,'value_name',value_name)) attrs
    from ${APP}.ods_sku_attr_value
    where dt='$do_date'
    group by sku_id
),
sale_attr as
(
    select
        sku_id,
        collect_set(named_struct('sale_attr_id',sale_attr_id,'sale_attr_value_id',sale_attr_value_id,'sale_attr_name',sale_attr_name,'sale_attr_value_name',sale_attr_value_name)) sale_attrs
    from ${APP}.ods_sku_sale_attr_value
    where dt='$do_date'
    group by sku_id
)

insert overwrite table ${APP}.dim_sku_info partition(dt='$do_date')
select
    sku.id,
    sku.price,
    sku.sku_name,
    sku.sku_desc,
    sku.weight,
    sku.is_sale,
    sku.spu_id,
    spu.spu_name,
    sku.category3_id,
    c3.name,
    c3.category2_id,
    c2.name,
    c2.category1_id,
    c1.name,
    sku.tm_id,
    tm.tm_name,
    attr.attrs,
    sale_attr.sale_attrs,
    sku.create_time
from sku
left join spu on sku.spu_id=spu.id
left join c3 on sku.category3_id=c3.id
left join c2 on c3.category2_id=c2.id
left join c1 on c2.category1_id=c1.id
left join tm on sku.tm_id=tm.id
left join attr on sku.id=attr.sku_id
left join sale_attr on sku.id=sale_attr.sku_id;
